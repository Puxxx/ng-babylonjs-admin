# git udpate

<!-- 检查当前文件状态 -->
git status (-s 简单浏览命令)

<!-- 跟踪新文件 -->
git add (new files here)

<!-- 暂存已修改文件 -->
git add (modified files here)

<!-- 提交更新 -->
git commit \<files> (-a 一起提交) (-m 简单描述语言) 

<!-- 移除文件 -->
rm \<files> (简单删除文件)
git rm \<files> (删除文件不能被恢复) (--cached 从 Git 仓库中删除，本地保留) (\* 正则匹配所有文件)
